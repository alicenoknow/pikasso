## pikasso language

'.' - end of expression

'#' - variable

e.g. #x = 5.

### operators
'+' - addition

'-' - subtraction

'*' - multiplication

'/' - division

'=' - assignment

'!=' - not equal

'==' - equal

'<=' - less or equal

'>=' - greater or equal

'<' - less

'>' - greater

'||' - or

'&&' - and

'!' - negation

### special values
true
false

### loops
for #elem in #start #end {

}

while #condition {

}

### drawing

bg #r #g #b

fill #r #g #b #a

border #r #g #b #width

circle #x #y #r

rect #x #y #w #h

line #x1 #y1 #x2 #y2
